## Fundamental Types
```rust
fn main() {
    // (1)
    let x: i32 = 42;

    // i16는 16비트 정수형으로, 표현 가능한 범위는 -32,768에서 32,767까지
    // 따라서 아래 코드는 문제 없다
    let y: i16 = x.try_into().unwrap();
    assert!(y == 42);

    // (2)
    let x: i32 = 66_000;

    // 패닉 발생
    let _y: i16 = x.try_into().expect("cnnot convert");

    // (3)
    let x: i32 = 32_768;

    // 32768은 i16로 표현할 수 없는 값이므로, -32768로 변환된다
    let y: i16 = x as i16;
    assert!(y == -32768);

    // (4)
    let x = 42i32;
    // 변환이 안전한 경우에도 into() 메서드를 사용해야 한다
    let y: i64 = x.into();
    assert!(y == 42);
}
```
